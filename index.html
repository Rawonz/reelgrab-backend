const API = "https://reelgrab-backend-production.up.railway.app"; function analiz() { const url = document.getElementById("urlInput").value.trim(); if (!url) return showErr("URL boşdur"); setProgress(40, "Analiz..."); fetch(API + "/analiz", { method: "POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({url}) }) .then(r => r.json()) .then(data => { document.getElementById("pTitle").textContent = data.baslik; document.getElementById("preview").classList.add("vis"); hideProgress(); }) .catch(e => { showErr(e.message); hideProgress(); }); } function indir() { const url = document.getElementById("urlInput").value.trim(); const mp3 = document.getElementById("opt-mp3").classList.contains("on"); fetch(API + "/download", { method: "POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ url, quality: kalite, mp3 }) }) .then(res => res.blob()) .then(blob => { const a = document.createElement("a"); a.href = URL.createObjectURL(blob); a.download = "video.mp4"; a.click(); }) .catch(e => showErr(e.message)); }
